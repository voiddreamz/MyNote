Sierpinski三角形 — 递归调用（必考！）

function sjx(A1, A2, A3, N1)          % 标准递归格式
    B1 = (A1+A2)/2;  B2 = (A1+A3)/2;  B3 = (A2+A3)/2;
    N2 = N1 - 1;                           % 终止条件控制
    if N2 > 0
        sjx(A1, B1, B2, N2);             % 递归：每次分3个新三角形
        sjx(B1, A2, B3, N2);
        sjx(B2, B3, A3, N2);
        fill(real([A1,A2,A3,A1]), imag([A1,A2,A3,A1]), 'g');
        hold on;
        fill(real([B1,B2,B3,B1]), imag([B1,B2,B3,B1]), 'w');
    end
end

Koch雪花 — 向量化写法

new = [0, 1i, sqrt(3)/2+1i/2, 0];
for k = 1:8
    old = new;
    diff = (old(2:end) - old(1:end-1)) / 3;
    n = length(old) - 1;
    new(1:4:4*n-3) = old(1:end-1);
    new(2:4:4*n-2) = old(1:end-1) + diff;
    new(3:4:4*n-1) = new(2:4:4*n-2) + diff*(0.5+0.5*sqrt(3)*1i);
    new(4:4:4*n) = old(1:end-1) + 2*diff;
    new(4*n+1) = old(end);
    plot(new); axis equal;
end

第5章 解常微分方程

把高阶ODE化为一阶组（必考）

例：y'' = -y  →  设 y₁=y, y₂=y'  →  y₁'=y₂, y₂'=-y₁

% 写法一：匿名函数（最常用）
F = @(t, y) [y(2); -y(1)]
[T, Y] = ode45(F, [0, 10], [0, 2])   % (时间范围, 初值)

% 写法二：矩阵形式
F = @(t, y) [0, 1; -1, 0] * y

% 写法三：子函数（复杂问题推荐）
function main
    [T, Y] = ode45(@myode, [0, 10], [0, 2]);
    plot(T, Y(:,1));       % Y(:,1)=y, Y(:,2)=y'
end
function ydot = myode(t, y)
    ydot = [y(2); -y(1)];
end

ode求解器速查

指令适用方法说明
ode45非刚性（首选）四/五阶RK最常用
ode23非刚性二/三阶RK低精度快速
ode113非刚性变阶Adams高精度光滑问题
ode15s刚性数值微分法刚性首选
ode23s刚性Rosenbrock低阶刚性
ode23t中度刚性梯形法则无数值阻尼
ode23tb刚性TR-BDF2替代ode15s
刚性方程：解的某些分量变化非常快而另一些变化很慢。用ode45会极慢 → 必须换ode15s或ode23s。

% 延拓区间：
solinit = bvpinit(sol, [a_new, b_new]);  % 利用旧解扩展区间

打靶法（Shooting Method）概念

打靶法解本征值：尝试一个本征值 → 作为初值问题求解 → 检查是否满足边界条件 → 不满足则调整本征值重试 → 反复迭代至误差小于容许值。本质：把边值问题转化为多轮初值问题。

附录：开卷考试解题模板（按题型分类）

考试时根据题目类型，找到对应模板，替换参数即可。每类模板给出了核心代码骨架。

模板A：力学题（抛体、振动、滑落、摆）

凡是涉及"运动轨迹""速度变化"的力学问题 → ODE + ode45

% ==== 力学题通用模板 ====
clear; close all;

% 1. 定义物理常数
g = 9.8; m = 1; k = 0.1;                        % 重力、质量、阻力系数（按题改）

% 2. 列运动方程，化为一阶ODE组
%    一般形式：y(1)=位置, y(2)=速度, y(3)=位置2, y(4)=速度2…
odefun = @(t,y) [ ...
    y(2);                                        % 位置1的导数 = 速度1
    合外力/m;                                    % 速度1的导数 = 加速度（根据题意改写）
    y(4);                                        % 位置2的导数 = 速度2
    合外力/m                                     % 速度2的导数
];

% 3. 设置事件（如果需要检测"落地""脱离"等特殊时刻）
opts = odeset('Events', @(t,y) deal(...
    事件条件表达式,    % 例如 y(3) 表示落地; y(3)-y(1)*tan(α) 表示落到斜坡
    1,                          % 1=触发后停止, 0=不停止
    -1));                       % -1=向下穿越触发, 1=向上, 0=双向

% 4. 用ode45求解
[T, Y] = ode45(odefun, [t_start, t_end], [初值1, 初值2, 初值3, 初值4], opts);

% 5. 提取需要的结果
Y(:,1)   % 位置1随时间变化
Y(:,2)   % 速度1随时间变化
Y(end,:) % 最终时刻的全部状态

% 6. 画图
figure;
plot(T, Y(:,1));                                 % 画位置-时间图
xlabel('t(s)'); ylabel('x(m)'); grid on;

常见变体：

  • 无阻力抛体 → 可直接用解析公式，不用ode45
  • 能量守恒问题(单摆、滑落) → 列能量方程→解出速度表达式→解析或fzero
  • 共振问题 → 直接用振幅公式 A=F₀/√[(k-mΩ²)²+(γΩ)²]
  • 变质量问题(链条滑落) → 列变质量牛顿方程→化ODE

模板B：光学题（干涉、衍射、光栅）

凡是涉及"光强分布""条纹"的问题 → 直接写光强公式 → 画曲线，不需要ODE

% ==== 光学题通用模板 ====
clear; close all;
lambda = 500e-9;                                 % 波长 (按题改)
L = 1;                                           % 观察距离
y = linspace(-0.01, 0.01, 2000);                 % 屏上坐标范围（按题调整）
theta = atan(y/L);                               % 衍射角

% ---- 双缝干涉 ----
d = 0.001;                                       % 双缝间距
k = 2*pi/lambda;
delta_phi = k * d * y / L;                       % 相位差 (远场近似)
I = 2 * (1 + cos(delta_phi));                    % I₁=I₂=1时的干涉光强

% ---- 单缝衍射 ----
a = 5e-6;                                        % 缝宽
beta = pi * a * sin(theta) / lambda;
I = (sin(beta)./beta).^2;                        % sinc²(β)，beta=0处加eps防除零

% ---- 光栅衍射 ----
a = 2e-6; b = 6e-6; d = a+b; N = 10;             % 缝宽、挡板宽、光栅常数、缝数
beta = pi*a*sin(theta)/lambda;
alpha_g = pi*d*sin(theta)/lambda;
I = (sin(beta)./beta).^2 .* (sin(N*alpha_g)./sin(alpha_g)).^2;

% 画图
figure;
plot(y*1000, I);                                 % 横轴用mm单位
xlabel('屏上位置(mm)'); ylabel('相对光强'); grid on;

陷阱提醒：sin(β)/β 在 β=0 处是 0/0 → 加 +1e-12 或用 sinc 函数。光栅的 sin(Nα)/sin(α) 在 α=mπ 处也要防除零。

模板C：电磁题（磁场、安培力、感应电动势）

核心公式：B=μ₀I/(2πr)（直导线），dF=I·dl×B（安培力），ε=-dΦ/dt（法拉第）

% ==== 电磁题通用模板 ====
clear; close all;
mu0 = 4*pi*1e-7;                                 % 真空磁导率（恒定常数）

% ---- 类型1：长直导线的磁场分布 ----
I = 1; a = 0.1;                                  % 电流、特征距离
x = linspace(-3*a, 3*a, 500);                    % 坐标范围
B = mu0*I ./ (2*pi*abs(x));                      % B=μ₀I/(2πr)，./是逐元素除
% 多根导线：B_total = B1 + B2 + …（矢量叠加，注意方向）

% ---- 类型2：安培力（含积分）----
I1 = 1; I2 = 0.5; d = 0.2;                       % 两电流、距离
f = @(phi) I2 * mu0*I1/(2*pi) * a*sin(phi) ./ (d + a*cos(phi));
F = integral(f, 0, pi);                          % 从0到π积分求总力

% ---- 类型3：感应电动势 ----
omega = 2*pi*10; t = linspace(0, 0.2, 500);
for j = 1:length(t)
    f_flux = @(x) 1 ./ (d + x*cos(omega*t(j)));  % 被积函数
    Phi(j) = mu0*I/(2*pi) * integral(f_flux, -a, a); % 计算磁通量
end
epsilon = -gradient(Phi, t(2)-t(1));             % ε=-dΦ/dt（用gradient求数值导）

figure; plot(t, epsilon);
xlabel('t(s)'); ylabel('ε(V)'); grid on;

关键函数：integral(被积函数, 下限, 上限) 做定积分。gradient(y, dx) 求数值导数（比diff更准）。

模板D：量子物理题（势阱、康普顿散射、德布罗意波）

量子题一般为直接套公式计算+画图，不需要ODE。关键在于单位换算（J↔eV, m↔nm↔pm）。

% ==== 量子题通用模板 ====
clear; close all;
h = 6.626e-34;  h_bar = 1.0546e-34;              % 普朗克常数、约化普朗克常数
c = 2.998e8;  m_e = 9.109e-31;  e = 1.602e-19;   % 光速、电子质量、电子电荷

% ---- 类型1：一维无限深势阱 ----
L = 1e-9;                                        % 阱宽（常给nm，要转m: ×1e-9）
n = 1:5;                                         % 要算的能级数
E_n = n.^2 * pi^2 * h_bar^2 / (2*m_e*L^2);       % 能级公式 Eₙ∝n²
x = linspace(0, L, 500);
psi = sqrt(2/L) * sin(n(1)*pi*x/L);              % 波函数 ψ=√(2/L)sin(nπx/L)
prob = psi.^2;                                    % 概率密度 |ψ|²

% ---- 类型2：康普顿散射 ----
lambda_C = h/(m_e*c);                            % 康普顿波长 ≈ 2.426pm
lambda_in = 0.071e-9;                            % 入射波长（按题给）
theta = linspace(0, pi, 500);
lambda_out = lambda_in + lambda_C*(1 - cos(theta)); % 散射公式
K_e = h*c*(1/lambda_in - 1./lambda_out) / e;      % 电子动能(eV)，/e把J→eV

% ---- 类型3：德布罗意波长 ----
U = 10000;                                       % 加速电压(V)
lambda = h / sqrt(2*m_e*e*U);                    % 非相对论。U>20kV需加相对论修正
fprintf('λ = %.2f pm\n', lambda*1e12);           % 1pm = 1e-12m

figure; plot(x*1e9, psi);                        % 横轴转nm单位(×1e9)
xlabel('x(nm)'); ylabel('ψ'); grid on;

单位换算必记：1nm=1e-9m, 1pm=1e-12m, 1eV=1.602e-19J, 1μT=1e-6T。画图时横轴通常用nm/pm/μm。

模板E：数据处理题（插值、拟合、FFT、求零点、求极值）

纯数学操作，不涉及物理方程。直接调MATLAB内置函数。

% ==== 数据处理通用模板 ====
clear; close all;

% ---- 插值 ----
x = [0,1,2,3,4];  y = [1,3,2,5,4];               % 已知数据点
xi = 0:0.1:4;                                    % 要插值的点
yi = interp1(x, y, xi, 'spline');               % 'spline'可换'linear''pchip''nearest'
plot(x, y, 'o', xi, yi, '-');                      % 原始点+插值曲线

% ---- 多项式拟合 ----
p = polyfit(x, y, 2);                            % 2次拟合，p=[a,b,c]表示ax²+bx+c
y_fit = polyval(p, xi);                          % 计算拟合值
plot(xi, y_fit);                                  % 画拟合曲线

% ---- 指数拟合（先取对数再polyfit）----
y_log = log(y);                                  % y=Ae^{Bx} → ln(y)=ln(A)+Bx
p_exp = polyfit(x, y_log, 1);                    % 线性拟合
A = exp(p_exp(2));  B = p_exp(1);                % 还原系数

% ---- 求零点 ----
f = @(x) x.^2 - 2;                               % 要求解的函数 f(x)=0
root = fzero(f, 1);                              % 从x=1附近开始找根
root2 = fzero(f, [0, 2]);                         % 或在[0,2]区间找（端点异号才有效）
all_roots = roots([1, 0, -3, 5]);                % 多项式所有根：x³-3x+5=0

% ---- 求极小值 ----
f2 = @(x) (x-3).^2 + 2;                           % 要最小化的函数
x_min = fminbnd(f2, 0, 5);                        % 在[0,5]内找极小值点

% ---- FFT ----
Fs = 1000;  t_fft = 0:1/Fs:1;                     % 采样频率1kHz，时长1s
signal = sin(2*pi*50*t_fft) + sin(2*pi*120*t_fft);% 50Hz+120Hz合成信号
Y = fft(signal);                                 % 做FFT
Pyy = abs(Y).^2 / length(Y);                     % 功率谱
f = Fs*(0:length(Y)/2)/length(Y);                % 频率轴（只取前半）
plot(f, Pyy(1:length(f)));                       % 画频谱
xlabel('f(Hz)'); ylabel('功率');
